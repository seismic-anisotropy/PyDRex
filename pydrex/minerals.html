 91def voigt_averages(minerals, weights):
 92    """Calculate elastic tensors as the Voigt averages of a collection of `mineral`s.
 93
 94    Args:
 95    - `minerals` — list of `pydrex.minerals.Mineral` instances storing orientations and
 96      fractional volumes of the grains within each distinct mineral phase
 97    - `weights` (dict) — dictionary containing weights of each mineral
 98      phase, as a fraction of 1, in keys named "<phase>_fraction",
 99      e.g. "olivine_fraction"
100
101    Raises a ValueError if the minerals contain an unequal number of grains or stored
102    texture results.
103
104    """
105    n_grains = minerals[0].n_grains
106    if not np.all([m.n_grains == n_grains for m in minerals[1:]]):
107        raise ValueError("cannot average minerals with unequal grain counts")
108    n_steps = len(minerals[0].orientations)
109    if not np.all([len(m.orientations) == n_steps for m in minerals[1:]]):
110        raise ValueError(
111            "cannot average minerals with variable-length orientation arrays"
112        )
113    if not np.all([len(m.fractions) == n_steps for m in minerals]):
114        raise ValueError(
115            "cannot average minerals with variable-length grain volume arrays"
116        )
117
118    elastic_tensors = {}
119
120    # TODO: Perform rotation directly on the 6x6 matrices, see Carcione 2007.
121    # This trick is implemented in cpo_elastic_tensor.cc in Aspect.
122    average_tensors = np.zeros((n_steps, 6, 6))
123    for i in range(n_steps):
124        for mineral in minerals:
125            for n in range(n_grains):
126                match mineral.phase:
127                    case _core.MineralPhase.olivine:
128                        if "olivine" not in elastic_tensors:
129                            elastic_tensors["olivine"] = (
130                                _tensors.voigt_to_elastic_tensor(OLIVINE_STIFFNESS)
131                            )
132                        average_tensors[i] += _tensors.elastic_tensor_to_voigt(
133                            _tensors.rotate(
134                                elastic_tensors["olivine"],
135                                mineral.orientations[i][n, ...].transpose(),
136                            )
137                            * mineral.fractions[i][n]
138                            * weights["olivine_fraction"]
139                        )
140                    case _core.MineralPhase.enstatite:
141                        if "enstatite" not in elastic_tensors:
142                            elastic_tensors["enstatite"] = (
143                                _tensors.voigt_to_elastic_tensor(ENSTATITE_STIFFNESS)
144                            )
145                        average_tensors[i] += _tensors.elastic_tensor_to_voigt(
146                            _tensors.rotate(
147                                elastic_tensors["enstatite"],
148                                minerals.orientations[i][n, ...].transpose(),
149                            )
150                            * mineral.fractions[i][n]
151                            * weights["enstatite_fraction"]
152                        )
153                    case _:
154                        raise ValueError(f"unsupported mineral phase: {mineral.phase}")
155    return average_tensors

158@dataclass
159class Mineral:
160    """Class for storing polycrystal texture for a single mineral phase.
161
162    A `Mineral` stores texture data for an aggregate of grains*.
163    Additionally, mineral fabric type and deformation regime are also tracked.
164    To provide an initial texture for the mineral, use the constructor arguments
165    `fractions_init` and `orientations_init`. By default,
166    a uniform volume distribution of random orientations is generated.
167
168    The `update_orientations` method computes new orientations and grain volumes
169    for a given velocity gradient. These results are stored in the `.orientations` and
170    `.fractions` attributes of the `Mineral` instance. The method also returns the
171    updated macroscopic deformation gradient based on the provided initial deformation
172    gradient.
173
174    *Note that the "number of grains" is a static integer value that
175    does not track the actual number of physical grains in the deforming polycrystal.
176    Instead, this number acts as a "number of bins" for the statistical resolution of
177    the crystallographic orientation distribution. The value is roughly equivalent to
178    (a multiple of) the number of initial, un-recrystallised grains in the polycrystal.
179    It is assumed that recrystallised grains do not grow large enough to require
180    rotation tracking.
181
182    **Examples:**
183
184    Mineral with isotropic initial texture:
185    >>> import pydrex
186    >>> olA = pydrex.Mineral(
187    ...     phase=pydrex.MineralPhase.olivine,
188    ...     fabric=pydrex.MineralFabric.olivine_A,
189    ...     regime=pydrex.DeformationRegime.dislocation,
190    ...     n_grains=2000
191    ... )
192    >>> olA.phase
193    <MineralPhase.olivine: 0>
194    >>> olA.fabric
195    <MineralFabric.olivine_A: 0>
196    >>> olA.regime
197    <DeformationRegime.dislocation: 1>
198    >>> olA.n_grains
199    2000
200
201    Mineral with specified initial texture and default phase, fabric and regime settings
202    which are for an olivine A-type mineral in the dislocation creep regime.
203    The initial grain volume fractions should be normalised.
204    >>> import numpy as np
205    >>> from scipy.spatial.transform import Rotation
206    >>> import pydrex
207    >>> rng = np.random.default_rng()
208    >>> n_grains = 2000
209    >>> olA = pydrex.Mineral(
210    ...     n_grains=n_grains,
211    ...     fractions_init=np.full(n_grains, 1 / n_grains),
212    ...     orientations_init=Rotation.from_euler(
213    ...         "zxz", [
214    ...             [x * np.pi / 2, np.pi / 2, np.pi / 2] for x in rng.random(n_grains)
215    ...         ]
216    ...     ).inv().as_matrix(),
217    ... )
218    >>> len(olA.orientations)
219    1
220    >>> type(olA.orientations)
221    <class 'list'>
222    >>> olA.orientations[0].shape
223    (2000, 3, 3)
224    >>> olA.fractions[0].shape
225    (2000,)
226
227    Note that minerals can also be constructed from serialized data,
228    see `Mineral.load` and `Mineral.from_file`.
229
230    **Attributes:**
231    - `phase` (`pydrex.core.MineralPhase`) — ordinal number of the mineral phase
232    - `fabric` (`pydrex.core.MineralFabric`) — ordinal number of the fabric type
233    - `regime` (`pydrex.core.DeformationRegime`) — ordinal number of the deformation
234      regime
235    - `n_grains` (int) — number of grains in the aggregate
236    - `fractions` (list of arrays) — grain volume fractions for each texture snapshot
237    - `orientations` (list of arrays) — grain orientation matrices for each texture
238      snapshot
239    - `seed` (int) — seed used by the random number generator to set up the isotropic
240      initial condition when `fractions_init` or `orientations_init` are not provided
241    - `lband` (int) — passed to the `scipy.integrate.LSODA` solver
242    - `uband` (int) — passed to the `scipy.integrate.LSODA` solver
243
244    """

288    def __post_init__(self):
289        """Initialise random orientations and grain volume fractions."""
290        if self.fractions_init is None:
291            self.fractions_init = np.full(self.n_grains, 1.0 / self.n_grains)
292        if self.orientations_init is None:
293            self.orientations_init = Rotation.random(
294                self.n_grains, random_state=self.seed
295            ).as_matrix()
296        # For large numbers of grains, the number of ODE's exceeds what LSODA can
297        # handle. Therefore, we specify the Jacobian matrix as banded.
298        # By default, we use a bandwidth o f 12000 with lband = uband = 6000.
299        # This should work for up to 10000 grains.
300        if self.lband is None and self.uband is None and self.n_grains > 4632:
301            _log.warning(
302                "using a banded Jacobian because of the large number of grains."
303                + " To manually control the bandwidth, set `lband` and/or `uband`"
304                + f" in calls to `{self.__class__.__qualname__}.update_orientations`."
305            )
306            self.lband = 6000
307            self.uband = 6000
308
309        # Copy the initial values to the storage lists.
310        self.fractions.append(self.fractions_init)
311        self.orientations.append(self.orientations_init)
312
313        # Delete the initial value duplicates to avoid confusion.
314        del self.fractions_init
315        del self.orientations_init
316
317        _log.info("created %s", self)

344    def update_orientations(
345        self,
346        config,
347        deformation_gradient,
348        get_velocity_gradient,
349        pathline,
350        **kwargs,
351    ):
352        """Update orientations and volume distribution for the `Mineral`.
353
354        Update crystalline orientations and grain volume distribution
355        for minerals undergoing plastic deformation. Return the updated deformation
356        gradient measuring the corresponding macroscopic deformation.
357
358        Args:
359        - `config` (dict) — PyDRex configuration dictionary
360        - `deformation_gradient` (array) — 3x3 initial deformation gradient tensor
361        - `get_velocity_gradient` (function) — callable with signature f(t, x) that
362          returns a 3x3 velocity gradient matrix at time t and position x (3D vector)
363        - `pathline` (tuple) — tuple consisting of:
364            1. the time at which to start the CPO integration (t_start)
365            2. the time at which to stop the CPO integration (t_end)
366            3. callable with signature f(t) that returns the position of the mineral at
367                time t ∈ [t_start, t_end]
368
369        Any additional (optional) keyword arguments are passed to
370        `scipy.integrate.LSODA`.
371
372        Array values must provide a NumPy-compatible interface:
373        <https://numpy.org/doc/stable/user/whatisnumpy.html>
374
375        """
376
377        # ===== Set up callables for the ODE solver and internal processing =====
378
379        def eval_rhs(t, y):
380            """Evaluate right hand side of the D-Rex PDE."""
381            # assert not np.any(np.isnan(y)), y[np.isnan(y)].shape
382            position = get_position(t)
383            velocity_gradient = get_velocity_gradient(t, position)
384            # _log.debug(
385            #     "calculating CPO at %s (t=%e) with velocity gradient %s",
386            #     position,
387            #     t,
388            #     velocity_gradient.ravel(),
389            # )
390
391            if self.phase == _core.MineralPhase.olivine:
392                volume_fraction = config["olivine_fraction"]
393            elif self.phase == _core.MineralPhase.enstatite:
394                volume_fraction = config["enstatite_fraction"]
395            else:
396                raise ValueError(
397                    f"phase must be a valid `MineralPhase`, not {self.phase}"
398                )
399
400            strain_rate = (velocity_gradient + velocity_gradient.transpose()) / 2
401            strain_rate_max = np.abs(la.eigvalsh(strain_rate)).max()
402            deformation_gradient, orientations, fractions = _utils.extract_vars(
403                y, self.n_grains
404            )
405            # Uses nondimensional values of strain rate and velocity gradient.
406            orientations_diff, fractions_diff = _core.derivatives(
407                phase=self.phase,
408                fabric=self.fabric,
409                n_grains=self.n_grains,
410                orientations=orientations,
411                fractions=fractions,
412                strain_rate=strain_rate / strain_rate_max,
413                velocity_gradient=velocity_gradient / strain_rate_max,
414                stress_exponent=config["stress_exponent"],
415                deformation_exponent=config["deformation_exponent"],
416                nucleation_efficiency=config["nucleation_efficiency"],
417                gbm_mobility=config["gbm_mobility"],
418                volume_fraction=volume_fraction,
419            )
420            return np.hstack(
421                (
422                    (velocity_gradient @ deformation_gradient).flatten(),
423                    orientations_diff.flatten() * strain_rate_max,
424                    fractions_diff * strain_rate_max,
425                )
426            )
427
428        def perform_step(solver):
429            """Perform SciPy solver step and appropriate processing."""
430            message = solver.step()
431            if message is not None and solver.status == "failed":
432                raise _err.IterationError(message)
433            # _log.debug(
434            #     "%s step_size=%e", solver.__class__.__qualname__, solver.step_size
435            # )
436
437            deformation_gradient, orientations, fractions = _utils.extract_vars(
438                solver.y, self.n_grains
439            )
440            orientations, fractions = _utils.apply_gbs(
441                orientations,
442                fractions,
443                config["gbs_threshold"],
444                self.orientations[-1],
445                self.n_grains,
446            )
447            solver.y[9:] = np.hstack((orientations.flatten(), fractions))
448
449        # ===== Initialise and run the solver using the above callables =====
450
451        time_start, time_end, get_position = pathline
452        if not callable(get_velocity_gradient):
453            raise ValueError(
454                "unable to evaluate velocity gradient callable."
455                + " You must provide a callable with signature f(t, x)"
456                + " that returns a 3x3 matrix."
457            )
458        if not callable(get_position):
459            raise ValueError(
460                "unable to evaluate position callable."
461                + " You must provide a callable with signature f(t)"
462                + " that returns a 3-component array."
463            )
464        _log.debug(
465            "calculating CPO from %s (t=%s) to %s (t=%s)",
466            get_position(time_start),
467            time_start,
468            get_position(time_end),
469            time_end,
470        )
471        _log.debug("    with deformation gradient %s", deformation_gradient.ravel())
472        _log.debug(
473            "    with velocity gradient interpolated between %s and %s",
474            get_velocity_gradient(time_start, get_position(time_start)).ravel(),
475            get_velocity_gradient(time_end, get_position(time_end)).ravel()
476        )
477        _log.debug(
478            "    intermediate velocity gradient = %s",
479            get_velocity_gradient(
480                (time_start + time_end) / 2, get_position((time_start + time_end) / 2)
481            ).ravel(),
482        )
483
484        y_start = np.hstack(
485            (
486                deformation_gradient.flatten(),
487                self.orientations[-1].flatten(),
488                self.fractions[-1],
489            )
490        )
491        solver = LSODA(
492            eval_rhs,
493            time_start,
494            y_start,
495            time_end,
496            atol=kwargs.pop("atol", np.abs(y_start * 1e-6) + 1e-12),
497            rtol=kwargs.pop("rtol", 1e-6),
498            first_step=kwargs.pop("first_step", np.abs(time_end - time_start) * 1e-1),
499            # max_step=kwargs.pop("max_step", np.abs(time_end - time_start)),
500            lband=self.lband,
501            uband=self.uband,
502            **kwargs,
503        )
504        perform_step(solver)
505        while solver.status == "running":
506            perform_step(solver)
507
508        # Extract final values for this simulation step, append to storage.
509        deformation_gradient, orientations, fractions = _utils.extract_vars(
510            solver.y.squeeze(), self.n_grains
511        )
512        self.orientations.append(orientations)
513        self.fractions.append(fractions)
514        return deformation_gradient

516    def save(self, filename, postfix=None):
517        """Save CPO data for all stored timesteps to a `numpy` NPZ file.
518
519        If `postfix` is not `None`, the data is appended to the NPZ file
520        in fields ending with "`_postfix`".
521
522        Raises a `ValueError` if the data shapes are not compatible.
523
524        See also: `numpy.savez`, `Mineral.load`, `Mineral.from_file`.
525
526        """
527        if len(self.fractions) != len(self.orientations):
528            raise ValueError(
529                "Length of stored results must match."
530                + " You've supplied currupted data with:\n"
531                + f"- {len(self.fractions)} grain size results, and\n"
532                + f"- {len(self.orientations)} orientation results."
533            )
534        if self.fractions[0].shape[0] == self.orientations[0].shape[0] == self.n_grains:
535            data = {
536                "meta": np.array(
537                    [self.phase, self.fabric, self.regime], dtype=np.uint8
538                ),
539                "fractions": np.stack(self.fractions),
540                "orientations": np.stack(self.orientations),
541            }
542            # Create parent directories, resolve relative paths.
543            _io.resolve_path(filename)
544            # Append to file, requires postfix (unique name).
545            if postfix is not None:
546                _log.info("saving Mineral to file %s (postfix: %s)", filename, postfix)
547                archive = ZipFile(filename, mode="a", allowZip64=True)
548                for key in data.keys():
549                    with archive.open(
550                        f"{key}_{postfix}", "w", force_zip64=True
551                    ) as file:
552                        buffer = io.BytesIO()
553                        np.save(buffer, data[key])
554                        file.write(buffer.getvalue())
555                        buffer.close()
556            else:
557                _log.info("saving Mineral to file %s", filename)
558                np.savez(filename, **data)
559        else:
560            raise ValueError(
561                "Size of CPO data arrays must match number of grains."
562                + " You've supplied corrupted data with:\n"
563                + f"- `n_grains = {self.n_grains}`,\n"
564                + f"- `fractions[0].shape = {self.fractions[0].shape}`, and\n"
565                + f"- `orientations[0].shape = {self.orientations[0].shape}`."
566            )

568    def load(self, filename, postfix=None):

570
571        If `postfix` is not `None`, data is read from fields ending with "`_postfix`".

Class for storing polycrystal texture for a single mineral phase.

A Mineral stores texture data for an aggregate of grains*. Additionally, mineral fabric type and deformation regime are also tracked. To provide an initial texture for the mineral, use the constructor arguments fractions_init and orientations_init. By default, a uniform volume distribution of random orientations is generated.

The update_orientations method computes new orientations and grain volumes for a given velocity gradient. These results are stored in the .orientations and .fractions attributes of the Mineral instance. The method also returns the updated macroscopic deformation gradient based on the provided initial deformation gradient.

*Note that the "number of grains" is a static integer value that does not track the actual number of physical grains in the deforming polycrystal. Instead, this number acts as a "number of bins" for the statistical resolution of the crystallographic orientation distribution. The value is roughly equivalent to (a multiple of) the number of initial, un-recrystallised grains in the polycrystal. It is assumed that recrystallised grains do not grow large enough to require rotation tracking.
