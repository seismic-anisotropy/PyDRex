pydrex.diagnostics

PyDRex: Methods to calculate texture and strain diagnostics.

Calculations expect orientation matrices $a$ to represent passive (i.e. alias) rotations, which are defined in terms of the extrinsic ZXZ euler angles $ϕ, θ, φ$ as $$ a = \begin{bmatrix} \cosφ\cosϕ - \cosθ\sinϕ\sinφ & \cosθ\cosϕ\sinφ + \cosφ\sinϕ & \sinφ\sinθ \cr -\sinφ\cosϕ - \cosθ\sinϕ\cosφ & \cosθ\cosϕ\cosφ - \sinφ\sinϕ & \cosφ\sinθ \cr \sinθ\sinϕ & -\sinθ\cosϕ & \cosθ \end{bmatrix} $$ such that a[i, j] gives the direction cosine of the angle between the i-th grain axis and the j-th external axis (in the global Eulerian frame).

 30def anisotropy(voigt_matrix, proj="hex"):
 31    """Calculate anisotropy diagnostics for the given elasticity tensor.
 32
 33    Args:
 34    - `voigt_matrix` (6x6 array) — the Voigt matrix representation of the averaged
 35      elasticity tensor for a polycrystal
 36    - `proj` — the projection function to use, currently only "hex" for hexagonal
 37      symmetry (transverse isotropy)
 38
 39    Returns the percent anisotropy, a matrix containing the axes of the Symmetry
 40    Cartesian Coordinate System (rows), and the elasticity tensor in the SCCS frame
 41    (Voigt vector representation).
 42
 43    """
 44    match proj:
 45        case "hex":
 46            proj = _tensors.hex_projector
 47        case _:
 48            raise ValueError(f"unsupported projector '{proj}'")
 49
 50    voigt_matrix = _tensors.upper_tri_to_symmetric(voigt_matrix)
 51    stiffness_dilat, stiffness_voigt = _tensors.voigt_decompose(voigt_matrix)
 52    # Appendix A5, Browaeys & Chevrot, 2004.
 53    K = np.trace(stiffness_dilat) / 9  # Bulk modulus
 54    G = (np.trace(stiffness_voigt) - 3 * K) / 10  # Shear modulus
 55    isotropic_vector = np.hstack(
 56        (
 57            np.repeat(K + 4 * G / 3, 3),
 58            np.repeat(np.sqrt(2) * (K - 2 * G / 3), 3),
 59            np.repeat(2 * G, 3),
 60            np.repeat(0, 12),
 61        )
 62    )
 63    voigt_vector = _tensors.voigt_matrix_to_vector(voigt_matrix)
 64    percent_anisotropy = (
 65        (la.norm(voigt_vector - isotropic_vector) - proj(voigt_vector))
 66        / la.norm(voigt_vector)
 67        * 100
 68    )
 69    eigv_dij = la.eigh(stiffness_dilat)[1][:, ::-1]
 70    eigv_vij = la.eigh(stiffness_voigt)[1][:, ::-1]
 71
 72    # Search for SCCA directions.
 73    for i in range(3):
 74        ndvc = 0
 75        advc = 10
 76        for j in range(3):
 77            sdv = np.clip(np.dot(eigv_dij[:, i], eigv_vij[:, j]), -1, 1)
 78            adv = np.arccos(abs(sdv))
 79            if adv < advc:
 80                ndvc = int((j + 1) * np.sign(sdv)) if sdv != 0 else j + 1
 81                advc = adv
 82        eigv_dij[:, i] = (eigv_dij[:, i] + ndvc * eigv_vij[:, abs(ndvc) - 1]) / 2
 83        eigv_dij[:, i] /= la.norm(eigv_dij[:, i], ord=2)
 84
 85    # Higher symmetry axis
 86    elastic_tensor = _tensors.voigt_to_elastic_tensor(voigt_matrix)
 87    voigt_norm = la.norm(voigt_vector)
 88    for i in range(3):
 89        dev = proj(
 90            _tensors.voigt_matrix_to_vector(
 91                _tensors.elastic_tensor_to_voigt(
 92                    _tensors.rotate(
 93                        elastic_tensor,
 94                        eigv_dij[:, [(i + j) % 3 for j in range(3)]].transpose(),
 95                    )
 96                )
 97            )
 98        )
 99        if dev < voigt_norm:
100            voigt_norm = dev
101            ndvc = i + 1
102
103    # Rotate in SCCA
104    sccs = eigv_dij[:, [(abs(ndvc) - 1 + i) % 3 for i in range(3)]].transpose()
105    voigt_vector = _tensors.voigt_matrix_to_vector(
106        _tensors.elastic_tensor_to_voigt(_tensors.rotate(elastic_tensor, sccs))
107    )
108    return percent_anisotropy, sccs / la.norm(sccs, axis=1)[:, None], voigt_vector

111def bingham_average(orientations, axis="a"):
112    """Compute Bingham averages from olivine orientation matrices.
113
114    Returns the antipodally symmetric average orientation
115    of the given crystallographic `axis`, or the a-axis by default.
116    Valid axis specifiers are "a" for [100], "b" for [010] and "c" for [001].
117
118    See also: [Watson 1966](https://doi.org/10.1086%2F627211),
119    [Mardia & Jupp, “Directional Statistics”](https://doi.org/10.1002/9780470316979).
120
121
122    """
123    match axis:
124        case "a":
125            row = 0
126        case "b":
127            row = 1
128        case "c":
129            row = 2
130        case _:
131            raise ValueError(f"axis must be 'a', 'b', or 'c', not {axis}")
132
133    # https://courses.eas.ualberta.ca/eas421/lecturepages/orientation.html
134    # Eigenvector corresponding to largest eigenvalue is the mean direction.
135    # SciPy returns eigenvalues in ascending order (same order for vectors).
136    # SciPy uses lower triangular entries by default, no need for all components.
137    mean_vector = la.eigh(_st._scatter_matrix(orientations, row))[1][:, -1]
138    return mean_vector / la.norm(mean_vector)

141def finite_strain(deformation_gradient, **kwargs):
142    """Extract measures of finite strain from the deformation gradient.
143
144    Extracts the vector defining the axis of maximum extension (longest semiaxis of the
145    finite strain ellipsoid) and the largest principal strain value from the 3x3
146    deformation gradient tensor.
147
148    """
149    # Get eigenvalues and eigenvectors of the left stretch (Cauchy-Green) tensor.
150    B_λ, B_v = la.eigh(
151        deformation_gradient @ deformation_gradient.transpose(),
152        driver=kwargs.pop("driver", "ev"),
153    )
154    # Stretch ratio is sqrt(λ) - 1, the (-1) is so that undeformed strain is 0 not 1.
155    return np.sqrt(B_λ[-1]) - 1, B_v[:, -1]

158def symmetry(orientations, axis="a"):
159    r"""Compute texture symmetry eigenvalue diagnostics from olivine orientation matrices.
160
161    Compute Point, Girdle and Random symmetry diagnostics
162    for ternary texture classification.
163    Returns the tuple (P, G, R) where
164    $$
165    \begin{align\*}
166    P &= (λ_{1} - λ_{2}) / N \cr
167    G &= 2 (λ_{2} - λ_{3}) / N \cr
168    R &= 3 λ_{3} / N
169    \end{align\*}
170    $$
171    with $N$ the sum of the eigenvalues $λ_{1} ≥ λ_{2} ≥ λ_{3}$
172    of the scatter (inertia) matrix.
173
174    See e.g. [Vollmer 1990](https://doi.org/10.1130/0016-7606(1990)102%3C0786:AAOEMT%3E2.3.CO;2).
175
176    """
177    match axis:
178        case "a":
179            row = 0
180        case "b":
181            row = 1
182        case "c":
183            row = 2
184        case _:
185            raise ValueError(f"axis must be 'a', 'b', or 'c', not {axis}")
186
187    scatter = _st._scatter_matrix(orientations, row)
188    # SciPy uses lower triangular entries by default, no need for all components.
189    eigvals_descending = la.eigvalsh(scatter)[::-1]
190    sum_eigvals = np.sum(eigvals_descending)
191    return (
192        (eigvals_descending[0] - eigvals_descending[1]) / sum_eigvals,
193        2 * (eigvals_descending[1] - eigvals_descending[2]) / sum_eigvals,
194        3 * eigvals_descending[2] / sum_eigvals,
195    )

198def misorientation_index(orientations, bins=None, system=(2, 4)):
199    """Calculate M-index for polycrystal orientations.
200
201    The `bins` argument is passed to `numpy.histogram`.
202    If left as `None`, 1° bins will be used as recommended by the reference paper.
203    The symmetry system can be specified using the `system` argument.
204    The default system is orthorhombic.
205
206    See [Skemer et al. 2005](https://doi.org/10.1016/j.tecto.2005.08.023).
207
208    """
209    # Compute and bin misorientation angles from orientation data.
210    combinations = np.empty((sp.comb(len(orientations), 2, exact=True), 2, 4))
211    for i, e in enumerate(
212        it.combinations(Rotation.from_matrix(orientations).as_quat(), 2)
213    ):
214        combinations[i] = list(e)
215    _log.debug("Largest array size: %s GB", combinations.nbytes / 1e9)
216
217    misorientations_data = misorientation_angles(combinations)
218    θmax = _st._max_misorientation(system)
219    misorientations_count, bin_edges = np.histogram(
220        misorientations_data, bins=θmax, range=(0, θmax), density=True
221    )
222
223    # Compute counts of theoretical misorientation for an isotropic aggregate,
224    # using the same bins (Skemer 2005 recommend 1° bins).
225    misorientations_theory = np.array(
226        [
227            _st.misorientations_random(bin_edges[i], bin_edges[i + 1])
228            for i in range(len(misorientations_count))
229        ]
230    )
231
232    # Equation 2 in Skemer 2005.
233    return (θmax / (2 * len(misorientations_count))) * np.sum(
234        np.abs(misorientations_theory - misorientations_count)
235    )

238def coaxial_index(orientations, axis1="b", axis2="a"):
239    r"""Calculate coaxial “BA” index for a combination of two crystal axes.
240
241    The BA index of [Mainprice et al. 2015](https://doi.org/10.1144/SP409.8)
242    is derived from the eigenvalue `symmetry` diagnostics and measures point vs girdle
243    symmetry in the aggregate. $BA ∈ [0, 1]$ where $BA = 0$ corresponds to a perfect
244    axial girdle texture and $BA = 1$ represents a point symmetry texture assuming that
245    the random component $R$ is negligible. May be less susceptible to random
246    fluctuations compared to the raw eigenvalue diagnostics.
247
248    """
249    P1, G1, _ = symmetry(orientations, axis=axis1)
250    P2, G2, _ = symmetry(orientations, axis=axis2)
251    return 0.5 * (2 - (P1 / (G1 + P1)) - (G2 / (G2 + P2)))

254def misorientation_angles(combinations):
255    """Calculate the misorientation angles for pairs of rotation quaternions.
256
257    Calculate the angular distance between the rotations `combinations[:, 0]`
258    and `combinations[:, 1]`, which are expected to be 1x4 passive (alias)
259    rotation quaternions.
260
261    Uses ~25% less memory than the same operation with rotation matrices.
262
263    See also:
264    - <https://math.stackexchange.com/questions/90081/quaternion-distance>
265    - <https://link.springer.com/article/10.1007/s10851-009-0161-2>
266
267
268    """
269    return 2 * np.rad2deg(
270        np.arccos(
271            np.abs(
272                np.clip(
273                    np.sum(combinations[:, 0] * combinations[:, 1], axis=1), -1.0, 1.0
274                )
275            )
276        )
277    )

280def smallest_angle(vector, axis):
281    """Get smallest angle between a unit `vector` and a bidirectional `axis`.
282
283    The axis is specified using either of its two parallel unit vectors.
284
285    """
286    angle = np.rad2deg(
287        np.arccos(np.dot(vector, axis) / (la.norm(vector) * la.norm(axis)))
288    )
289    if angle > 90:
290        return 180 - angle
291    return angle
