tests.test_simple_shear_2d

PyDRex: 2D simple shear tests.

NOTE: In scipy, rotations are represented as a matrix that transforms [1, 0, 0] to the new a-axis vector. We need the inverse of that rotation, which represents the change of coordinates from the grain-local to the global (Eulerian) frame.

 26    def test_shearYZ_initQ1(
 27        self,
 28        params_Kaminski2001_fig5_solid,  # GBM = 0
 29        params_Kaminski2001_fig5_shortdash,  # GBM = 50
 30        params_Kaminski2001_fig5_longdash,  # GBM = 200
 31        rng,
 32        outdir,
 33    ):
 34        """Test clockwise a-axis rotation around X.
 35
 36        Initial condition: randomised a-axis orientation within the first
 37        quadrant (between +Y and +Z axes) and steady flow with velocity gradient
 38
 39                0 0 0
 40            L = 0 0 2
 41                0 0 0
 42
 43        Orientations set up for slip on (010)[100].
 44        Tests the effect of grain boundary migration,
 45        similar to Fig. 5 in [Kaminski 2001].
 46
 47        [Kaminski 2001]: https://doi.org/10.1016%2Fs0012-821x%2801%2900356-9
 48
 49        """
 50        strain_rate_scale = 2e-5
 51        velocity_gradient = np.zeros((3, 3))
 52        velocity_gradient[1, 2] = strain_rate_scale
 53        timescale = 1 / (strain_rate_scale / 2)
 54        n_grains = 1000
 55
 56        # Initial orientations with a-axis in first quadrant of the YZ plane,
 57        # and c-axis along the +X direction (important!)
 58        # This means that the starting average is the same,
 59        # which is convenient for comparing the texture evolution.
 60        orientations_init = (
 61            Rotation.from_euler(
 62                "zxz",
 63                [[x * np.pi / 2, np.pi / 2, np.pi / 2] for x in rng.random(n_grains)],
 64            )
 65            .inv()
 66            .as_matrix()
 67        )
 68        # Uncomment to check a-axis vectors, should be near [0, a, a].
 69        # assert False, f"{orientations_init[0:10, 0, :]}"
 70        # Uncomment to check c-axis vectors, should be near [1, 0, 0].
 71        # assert False, f"{orientations_init[0:10, 2, :]}"
 72
 73        # One mineral to test each value of grain boundary mobility.
 74        minerals = [
 75            _minerals.Mineral(
 76                _minerals.MineralPhase.olivine,
 77                _minerals.OlivineFabric.A,
 78                _defmech.Regime.dislocation,
 79                n_grains=n_grains,
 80                fractions_init=np.full(n_grains, 1 / n_grains),
 81                orientations_init=orientations_init,
 82            )
 83            for i in range(3)
 84        ]
 85
 86        # Optional plotting setup.
 87        if outdir is not None:
 88            labels = []
 89            angles = []
 90            indices = []
 91
 92        for mineral, params in zip(
 93            minerals,
 94            (
 95                params_Kaminski2001_fig5_solid,  # GBM = 0
 96                params_Kaminski2001_fig5_shortdash,  # GBM = 50
 97                params_Kaminski2001_fig5_longdash,  # GBM = 200
 98            ),
 99        ):
100            time = 0
101            timestep = 0.025
102            timestop = 1
103            deformation_gradient = np.eye(3)  # Undeformed initial state.
104            while time < timestop * timescale:
105                deformation_gradient = mineral.update_orientations(
106                    params,
107                    deformation_gradient,
108                    velocity_gradient,
109                    integration_time=timestep * timescale,
110                )
111                time += timestep * timescale
112
113            n_timesteps = len(mineral.orientations)
114            misorient_angles = np.zeros(n_timesteps)
115            misorient_indices = np.zeros(n_timesteps)
116            # Loop over first dimension (time steps) of orientations.
117            for idx, matrices in enumerate(mineral.orientations):
118                orientations_resampled, _ = _stats.resample_orientations(
119                    matrices, mineral.fractions[idx]
120                )
121                direction_mean = _diagnostics.bingham_average(
122                    orientations_resampled,
123                    axis=_minerals.get_primary_axis(mineral.fabric),
124                )
125                misorient_angles[idx] = _diagnostics.smallest_angle(
126                    direction_mean, [0, 1, 0]
127                )
128                misorient_indices[idx] = _diagnostics.misorientation_index(
129                    orientations_resampled
130                )
131
132            # Check for uncorrupted record of initial condition.
133            assert np.isclose(misorient_angles[0], 45.0, atol=5.0)
134            assert misorient_indices[0] < 0.71
135            # Check for mostly smoothly decreasing misalignment.
136            angles_diff = np.diff(misorient_angles)
137            assert np.max(angles_diff) < 3.6
138            assert np.min(angles_diff) > -7.5
139            assert np.sum(angles_diff) < -24.0
140
141            # Check alignment and texture strength (half way and final value).
142            halfway = round(n_timesteps / 2)
143            match params["gbm_mobility"]:
144                case 0:
145                    np.testing.assert_allclose(
146                        misorient_angles,
147                        misorient_angles[0]
148                        * np.exp(
149                            np.linspace(0, timestop, n_timesteps)
150                            * (np.cos(2 * np.deg2rad(misorient_angles[0])) - 1)
151                        ),
152                        atol=5.0,
153                    )
154                    assert np.isclose(misorient_angles[halfway], 25, atol=2.0)
155                    assert np.isclose(misorient_angles[-1], 17.0, atol=1.0)
156                    assert np.isclose(misorient_indices[halfway], 0.925, atol=0.075)
157                    assert np.isclose(misorient_indices[-1], 0.975, atol=0.05)
158                case 50:
159                    assert np.isclose(misorient_angles[halfway], 15, atol=2.0)
160                    assert np.isclose(misorient_angles[-1], 10, atol=1.0)
161                    assert np.isclose(misorient_indices[halfway], 0.925, atol=0.075)
162                    assert np.isclose(misorient_indices[-1], 0.97, atol=0.03)
163                case 200:
164                    assert np.isclose(misorient_angles[halfway], 9, atol=1.5)
165                    assert np.isclose(misorient_angles[-1], 7, atol=1.0)
166                    assert np.isclose(misorient_indices[halfway], 0.975, atol=0.05)
167                    assert np.isclose(misorient_indices[-1], 0.99, atol=0.03)
168
169            # Optionally store plotting metadata.
170            if outdir is not None:
171                labels.append(f"$M^$ = {params['gbm_mobility']}")
172                angles.append(misorient_angles)
173                indices.append(misorient_indices)
174
175        # Optionally plot figure.
176        if outdir is not None:
177            _vis.simple_shear_stationary_2d(
178                angles,
179                indices,
180                timestop=timestop,
181                savefile=f"{outdir}/simple_shearYZ_stationary_olivineA_initQ1.png",
182                markers=("o", "v", "s"),
183                labels=labels,
184                refval=45,
185            )
