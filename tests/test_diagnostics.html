116    def test_upsample(self):
117        """Test upsampling of the raw orientation data."""
118        orientations = (

134        )

158    def test_common_input_errors(self):
159        """Test that exceptions are raised for bad input data."""
160        orientations = Rotation.from_rotvec(
161            [
162                [0, 0, 0],
163                [0, 0, np.pi / 6],
164                [np.pi / 6, 0, 0],
165            ]
166        )
167        fractions = np.array([0.25, 0.6, 0.15])
168        with pytest.raises(ValueError):
169            # SciPy Rotation instance is not valid input.
170            _stats.resample_orientations(orientations, fractions)
171            # Input must be a stack of orientations.
172            _stats.resample_orientations(orientations.as_matrix(), fractions)
173            # Input must be a stack of fractions.
174            _stats.resample_orientations([orientations.as_matrix()], fractions)
175            # First two dimensions of inputs must match.
176            _stats.resample_orientations([orientations.as_matrix()], [fractions[:-1]])
177
178        # This is the proper way to do it:
179        _stats.resample_orientations([orientations.as_matrix()], [fractions])

185    def test_average_0(self):
186        """Test bingham average of vectors aligned to the reference frame."""
187        orientations = Rotation.from_rotvec([[0, 0, 0]] * 10).inv().as_matrix()

189        b_mean = _diagnostics.bingham_average(orientations, axis="b")
190        c_mean = _diagnostics.bingham_average(orientations, axis="c")
191        np.testing.assert_array_equal(a_mean, [1, 0, 0])
192        np.testing.assert_array_equal(b_mean, [0, 1, 0])
193        np.testing.assert_array_equal(c_mean, [0, 0, 1])

206        )

215        """Test bingham average of vectors spread within 10 of the X-axis."""

225            .inv()
226            .as_matrix()
227        )
228        a_mean = _diagnostics.bingham_average(orientations, axis="a")
229        b_mean = _diagnostics.bingham_average(orientations, axis="b")
230        c_mean = _diagnostics.bingham_average(orientations, axis="c")
